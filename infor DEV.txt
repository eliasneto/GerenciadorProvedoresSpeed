
# Criar maquina Virtual:
python3 -m venv .venv

# Rode este comando para instalar o Django e o REST Framework (essencial para sistemas de gestão):
pip install django djangorestframework python-decouple

# Criar o Projeto Principal
python -m django startproject config .

# Criar a pasta para os Módulos
mkdir apps

# Criar o Módulo Core
python manage.py startapp core apps/core

# 1. Cria os arquivos de migração para o módulo core
python manage.py makemigrations core

# 2. Aplica as mudanças no banco de dados
python manage.py migrate

#Clonar um repositorio:
git clone -b dev https://github.com/eliasneto/GerenciadorProvedoresSpeed.git

#Criar arquivo requirements:
pip freeze > requirements.txt

#Instalar apartir do arquivo requirements:
pip install -r requirements.txt

# Remover um Container:
docker rm -f ID_OU_NOME

# lista e remove imagens:
docker images
docker rmi ID_DA_IMAGEM

#Lista e exclui Volumes:
docker volume ls  # Para ver o nome do volume
docker volume rm NOME_DO_VOLUME

# Comando para excluir pasta com arquivos dentro no linux: 
rm -rf nome_da_pasta

# subindo docker limpando o cache: 
docker-compose down

docker logs -f speed_sistema

# Comando no docker:

docker exec -it speed_sistema python manage.py makemigrations
